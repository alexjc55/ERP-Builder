    FROM data d
    JOIN entity_records o
      ON o.entity_id = (SELECT id FROM entities WHERE entity_key = 'orders')
     AND o.values_json->>'order_number' = d.ord
    JOIN record_links l
      ON l.target_record_id = o.id
     AND l.relation_id = (SELECT id FROM relations WHERE relation_key = 'zakazy'
          AND source_entity_id = (SELECT id FROM entities WHERE entity_key = 'items')
          AND target_entity_id = (SELECT id FROM entities WHERE entity_key = 'orders'))
    JOIN entity_records it
      ON it.id = l.source_record_id
     AND it.entity_id = (SELECT id FROM entities WHERE entity_key = 'items')
     AND regexp_replace(it.values_json->>'item_name', '\s+', ' ', 'g') = d.iname
  )
  UPDATE entity_records er
  SET values_json = er.values_json || t.vals,
      updated_at = now()
  FROM tgt t
  WHERE er.id = t.id
    AND (er.values_json || t.vals) IS DISTINCT FROM er.values_json;
